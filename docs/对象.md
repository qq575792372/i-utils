### getValueByPath()

```ts
function getValueByPath(target: Record<any, any>, path: string): any;
```

根据字符串属性路径获取目标对象的值

#### 参数

##### target

`Record`\<`any`, `any`\>

目标对象

##### path

`string` = `"data"`

字符串属性路径

#### 返回

`any`

返回目标对象的值

#### 示例

```ts
let res = {code:200, data:{rows:[], pages:{current:1,pageSize:20}}}
this._getValueByPath(res, 'data.pages.pageSize'); // 这里会输出20
```

***

### setValueByPath()

```ts
function setValueByPath(
   target: Record<any, any>, 
   path: string, 
value: any): Record<any, any>;
```

根据字符串属性路径设置目标对象的值

#### 参数

##### target

`Record`\<`any`, `any`\>

目标对象

##### path

`string` = `"data"`

字符串属性路径

##### value

`any`

值

#### 返回

`Record`\<`any`, `any`\>

返回修改过后的目标对象

#### 示例

```ts
let res = {code:200, data:{rows:[], pages:{current:1,pageSize:20}}}
this._setValueByPath(res, 'data.pages.pageSize', 30); // 打印res对象会发现pageSize的值改为了30
```
